# Setup
```sh
npm i fashion
```
# Usage
```js
const fashion = require('fashion')

console.log(`${fashion.blue('Im blue!')}`)
console.log(`${fashion.yellow.bold.bgGreen('I am yellow with bold text on a green background!')}`)
```
